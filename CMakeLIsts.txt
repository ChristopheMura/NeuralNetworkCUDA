cmake_minimum_required(VERSION 3.20)
project(cuda_project LANGUAGES CXX CUDA)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CUDA_STANDARD 17)
set(CMAKE_CUDA_ARCHITECTURES 89)  # RTX 4060 = SM 8.9

# Trouver Python 3 et Numpy
find_package(Python3 COMPONENTS Interpreter Development NumPy REQUIRED)

execute_process(
    COMMAND ${Python3_EXECUTABLE} -c "import numpy; print(numpy.get_include())"
    OUTPUT_VARIABLE NumPy_INCLUDE_DIR
    OUTPUT_STRIP_TRAILING_WHITESPACE
    RESULT_VARIABLE NumPy_RESULT
)

# Désactive l'avertissement sur les archi dépréciées
set(CMAKE_CUDA_FLAGS "${CMAKE_CUDA_FLAGS} -Wno-deprecated-gpu-targets")

add_executable(run src/cuda_kernels.cu src/mylib.cpp src/main.cpp)

target_include_directories(run PRIVATE 
    ${Python3_INCLUDE_DIRS}
    ${NumPy_INCLUDE_DIR}
    ${CMAKE_CURRENT_SOURCE_DIR}/include
    ${CMAKE_CURRENT_SOURCE_DIR}/matplotlib-cpp  # Ajout pour matplotlib-cpp
)

# Permet de compiler avec MSVC récent (19.43)
set_target_properties(run PROPERTIES
    CUDA_SEPARABLE_COMPILATION ON
)

target_compile_options(run PRIVATE
    $<$<COMPILE_LANGUAGE:CUDA>:--allow-unsupported-compiler>
)

target_link_libraries(run PRIVATE ${Python3_LIBRARIES})
